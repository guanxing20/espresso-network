18pub trait DecodeRevertError {
19    fn maybe_decode_revert<E: SolInterface + std::fmt::Debug>(
20        self,
21    ) -> anyhow::Result<PendingTransactionBuilder<Ethereum>>;
22}
23
24impl DecodeRevertError
25    for alloy::contract::Result<PendingTransactionBuilder<Ethereum>, alloy::contract::Error>
26{
27    fn maybe_decode_revert<E: SolInterface + std::fmt::Debug>(
28        self,
29    ) -> anyhow::Result<PendingTransactionBuilder<Ethereum>> {
30        match self {
31            Ok(ret) => Ok(ret),
32            Err(err) => {
33                let decoded = err.as_decoded_interface_error::<E>();
34                let msg = match decoded {
35                    Some(e) => format!("{:?}", e),
36                    None => format!("{:?}", err),
37                };
38                Err(anyhow::anyhow!(msg))
39            },
40        }
41    }
42}
